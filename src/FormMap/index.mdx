# FormMap

提供 Antd 组件的基础表单生成。

## 基本用法

必须由外部建立 Form, 并传入 FormMap 的 `form` 属性。

<Playground>
  {() => {
    class FormDemo extends React.Component {
      render() {
        const items = [{
          key: 'inputDemo',
          initialValue: 'nnecec',
          formItemConfigs:{
            label: '名称',
          },
        }, {
          key:'selectDemo',
          formItemConfigs:{
            label: '性别',
          },
          type: {
            name: 'Select',
            options: [{
              value: 0, mean: '女'
            }, {
              value: 1, mean: '男'
            }],
          }
        }]
        const handleSubmit = (e) => {
          e.preventDefault();
          this.props.form.validateFields((err, values) => {
              console.log('Received values of form: ', values);
          });
        }
        return (
          <React.Fragment>
            <FormMap items={items} form={this.props.form} />
            <Button onClick={handleSubmit}>确定</Button>
          </React.Fragment>
        )
      }
    }

    const Demo = Form.create()(FormDemo)

    return (
      <Demo />
    )
  }}
</Playground>
