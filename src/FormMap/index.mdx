# FormMap

## 基本用法

必须由外部建立 Form, 并传入 FormMap 的 `form` 属性。

<Playground>
  {() => {
    class FM extends React.Component {
      render() {
        const items = [{
          key: 'inputDemo',
          FormItemConfigs:{
            label: '名称',
            labelCol: { span: 6 },
            wrapperCol: { span: 14 },
          },
          initialValue: 'nnecec',
        }, {
          key:'selectDemo',
          type: 'select',
          FormItemConfigs:{
            label: '性别',
          }
        }]
        const handleSubmit = (e) => {
          e.preventDefault();
          this.props.form.validateFields((err, values) => {
              console.log('Received values of form: ', values);
          });
        }
        return (
          <React.Fragment>
            <FormMap items={items} form={this.props.form} />
            <Button onClick={handleSubmit}>确定</Button>
          </React.Fragment>
        )
      }
    }

    const FormMapDemo = Form.create()(FM)

    return (
      <FormMapDemo></FormMapDemo>
    )
  }}
</Playground>
