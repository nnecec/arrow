# ImageUpload

## 基本用法

## 校验项

可选校验图片格式、图片大小、图片尺寸、限制上传图片数量

<Playground>
  <ImageUpload 
    action="/api/image/upload"
    format="jpg"
    size={2}
    measure={{width: 640, height: 480}}
    length={5}
  ></ImageUpload>
</Playground>

## 支持 Form
<Playground>
  {() => {
    class IU extends React.Component {
      render() {
        const { getFieldDecorator } = this.props.form
        return (
          <Form>
            <Form.Item
              label="图片列表"
            >
              {
                getFieldDecorator('imageList', {
                  initialValue: ['https://zos.alipayobjects.com/rmsportal/jkjgkEfvpUPVyRjUImniVslZfWPnJuuZ.png']
                })(
                  <ImageUpload action="/api/image/upload" />
                )}
            </Form.Item>

          </Form>
        )
      }
    }
    const IUForm = Form.create()(IU)
    return (
      <IUForm />
    )
  }}
</Playground>
