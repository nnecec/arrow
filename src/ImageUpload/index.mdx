# ImageUpload

## 基本用法

<Playground>
  <ImageUpload action="/api/image/upload"></ImageUpload>
</Playground>

## 校验项

可选校验图片格式、图片大小、图片尺寸、限制上传图片数量

<Playground>
  <ImageUpload 
    action="/api/image/upload"
    format="jpg"
    size={2}
    measure={{width: 640, height: 480}}
    length={5}
  ></ImageUpload>
</Playground>

## 支持Form
<Playground>
  {() => {
    class IU extends React.Component {
      render() {
        const { getFieldDecorator } = this.props.form
        return (
          <Form.Item
      >
        {
          getFieldDecorator('imageList', {
          })(
            <ImageUpload></ImageUpload>
          )}
      </Form.Item>
        )
      }
    }

    const ImageUploadDemo = Form.create()(IU)


    return (
      <ImageUploadDemo />
    )
  }}
</Playground>
